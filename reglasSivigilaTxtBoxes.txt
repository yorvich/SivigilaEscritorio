IF OCCURS('0',THIS.applicableRules)>0 THEN
	*El objeto debe ser sometido a la regla 0 que limita sus valores admisibles a un conjunto
	IF OCCURS(',' + valueToValidate + ',', ',' + THIS.referenceValues + ',' )=0 THEN
	   MESSAGEBOX( 'Digite un valor del conjunto de valors admitidos para el ítem', 0 + 48 + 256, 'Violación de validación SIVIGILA' )
	   RETURN 0
	ENDIF
ENDIF

IF OCCURS('1',THIS.applicableRules)>0 THEN
	*El objeto debe ser sometido a la regla 1 que limita sus valores admisibles a un rango

	IF EMPTY(valueToValidate) THEN
		IF INT(this.initValue)=this.initValue THEN
			MESSAGEBOX( 'Digite un valor entre '+ALLTRIM(STR(this.initValue))+' y '+ALLTRIM(STR(this.endValue)), 0 + 48 + 256, 'Violación de validación SIVIGILA' )
		ELSE
			MESSAGEBOX( 'Digite un valor entre '+ALLTRIM(STR(this.initValue,11,3))+' y '+ALLTRIM(STR(this.endValue,11,3)), 0 + 48 + 256, 'Violación de validación SIVIGILA' )
		ENDIF
		RETURN 0
	ENDIF

	IF !BETWEEN(val(valueToValidate),THIS.initValue,THIS.endValue) THEN
		IF INT(this.initValue)=this.initValue THEN
			MESSAGEBOX( 'Digite un valor entre '+ALLTRIM(STR(this.initValue))+' y '+ALLTRIM(STR(this.endValue)), 0 + 48 + 256, 'Violación de validación SIVIGILA' )
		ELSE
			MESSAGEBOX( 'Digite un valor entre '+ALLTRIM(STR(this.initValue,11,3))+' y '+ALLTRIM(STR(this.endValue,11,3)), 0 + 48 + 256, 'Violación de validación SIVIGILA' )
		ENDIF
		RETURN 0
	ENDIF
ENDIF

IF OCCURS('2',THIS.applicableRules)>0 THEN
	*El objeto debe ser sometido a la regla 2 que exige un valor diferente de nulo

	IF EMPTY(valueToValidate) THEN
		MESSAGEBOX( 'Digite el texto Requerido', 0 + 48 + 256, 'Violación de validación SIVIGILA' )
		RETURN 0
	ENDIF
ENDIF

IF OCCURS('3',THIS.applicableRules)>0 THEN
	*El objeto debe ser sometido a la regla 3 que exige un valor mayor que 0

	IF !(VAL(valueToValidate)>0) THEN
		MESSAGEBOX( 'Digite un valor mayor de cero (0)', 0 + 48 + 256, 'Violación de validación SIVIGILA' )
		RETURN 0
	ENDIF
ENDIF

IF OCCURS('4',THIS.applicableRules)>0 THEN
	*El objeto debe ser sometido a la regla 4 que exige un valor anterior a la fecha del sistema

	IF valueToValidate>DATE() THEN
		MESSAGEBOX( 'La fecha digitada no puede ser mayor a la fecha del Sistema', 0 + 48 + 256, 'Violación de validación SIVIGILA' )
		valueToValidate=CTOD('')
		RETURN 0
	ENDIF
ENDIF

IF OCCURS('5',THIS.applicableRules)>0 THEN
	*El objeto debe ser sometido a la pseudo-regla 5 que elimina cualquier caracter espècial que contenga
	*Comentado por: Wilson Aguilar	07/09/2013
	*Actualmente descartada (en desuso) por cuanto el método la aplica automáticamente como parte inicial de la validación
	specialChars = "+.\/:*?',<>_()=" + '"' 
	valueToValidate=CHRTRAN(valueToValidate,specialChars,'')
	This.Value=valueToValidate
ENDIF

IF OCCURS('6',THIS.applicableRules)>0 THEN
	*El objeto debe ser sometido a la regla 6 que exige un valor mayor o igual que 0

	IF !(VAL(valueToValidate)>=0) THEN
		MESSAGEBOX( 'Digite un valor mayor o igual a cero (0)', 0 + 48 + 256, 'Violación de validación SIVIGILA' )
		RETURN 0
	ENDIF
ENDIF


IF OCCURS('7',THIS.applicableRules)>0 THEN
	*El objeto debe ser sometido a la regla 7 creada específicamente para validar tipos de documento de identidad

	valueToValidate=UPPER(valueToValidate)
	IF valueToValidate!='RC' and valueToValidate!='TI' and valueToValidate!='CC' and valueToValidate!='CE' ;
		and valueToValidate!='PA' and valueToValidate!='MS' and valueToValidate!='AS' THEN
		MESSAGEBOX( 'El tipo de documento de identidad es inválido', 0 + 48 + 256, 'Violación de validación SIVIGILA' )
		RETURN 0
	ELSE
		This.Value=valueToValidate
	ENDIF
ENDIF


IF OCCURS('8',THIS.applicableRules)>0 THEN
	*El objeto debe ser sometido a la regla 8 creada específicamente para validar horas
	IF !BETWEEN(VAL(SUBSTR(valueToValidate,1,2)),0,24)
	   MESSAGEBOX( 'La hora debe ser un valor entre 0 y 24', 0 + 48 + 256, 'Violación de validación SIVIGILA' )
	   RETURN 0
	ENDIF

	IF !BETWEEN(VAL(SUBSTR(valueToValidate,4,2)),0,59)

	   RETURN 0
	ENDIF
ENDIF
